    <button id="spinButton" onclick="spin()">Spin</button>
    <div id="result"></div>

    <script>
        let spinning = false;

        function spin() {
            if (spinning) return;

            const wheel = document.getElementById("wheel");
            const spinButton = document.getElementById("spinButton");
            const result = document.getElementById("result");

            spinning = true;
            spinButton.disabled = true;

            const rotations = Math.floor(Math.random() * 5000) + 5000; // Random rotations
            const degrees = rotations % 360; // Ending degree
            const segments = ["Reward 1", "Reward 2", "Reward 3", "Reward 4", "Reward 5", "Reward 6"];
            const segmentSize = 360 / segments.length;
            const selectedSegment = Math.floor(degrees / segmentSize);

            wheel.style.transform = `rotate(${rotations}deg)`;

            setTimeout(() => {
                spinning = false;
                spinButton.disabled = false;
                result.innerText = `You won: ${segments[segments.length - 1 - selectedSegment]}`;
            }, 4000);
        }
    </script>
